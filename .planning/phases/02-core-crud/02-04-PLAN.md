Create `frontend/src/api/notes.js` following the same pattern as `frontend/src/api/tasks.js`:

```javascript
const API_URL = process.env.REACT_APP_API_URL || 'http://localhost:5000';

export const getNotes = async (token) => {
  const res = await fetch(`${API_URL}/api/notes`, {
    headers: { 'Authorization': `Bearer ${token}` }
  });
  return res.json();
};

export const createNote = async (token, noteData) => {
  const res = await fetch(`${API_URL}/api/notes`, {
    method: 'POST',
    headers: {
      'Content-Type': 'application/json',
      'Authorization': `Bearer ${token}`
    },
    body: JSON.stringify(noteData)
  });
  return res.json();
};

export const updateNote = async (token, id, noteData) => {
  const res = await fetch(`${API_URL}/api/notes/${id}`, {
    method: 'PUT',
    headers: {
      'Content-Type': 'application/json',
      'Authorization': `Bearer ${token}`
    },
    body: JSON.stringify(noteData)
  });
  return res.json();
};

export const deleteNote = async (token, id) => {
  const res = await fetch(`${API_URL}/api/notes/${id}`, {
    method: 'DELETE',
    headers: { 'Authorization': `Bearer ${token}` }
  });
  return res.json();
};
```

Four named exports. Every function takes `token` as first argument.

Create `frontend/src/pages/NotesPage.js`. Use inline styles only — no Tailwind, no external CSS. Notes are simpler than tasks: just title and content, no status or priority.

**State:**
```javascript
const [notes, setNotes] = useState([]);
const [loading, setLoading] = useState(true);
const [error, setError] = useState('');
const [showCreateModal, setShowCreateModal] = useState(false);
const [editNote, setEditNote] = useState(null);
const [formData, setFormData] = useState({ title: '', content: '' });
```

**Load on mount:**
```javascript
useEffect(() => { loadNotes(); }, []);

const loadNotes = async () => {
  setLoading(true);
  const data = await getNotes(token);
  if (data.notes) setNotes(data.notes);
  else setError('Failed to load notes');
  setLoading(false);
};
```

**Create:** `createNote(token, formData)` -> `loadNotes()` -> close modal -> reset form.

**Edit:** Clicking Edit sets `editNote` and populates `formData`. `updateNote(token, editNote.id, formData)` -> `loadNotes()` -> close modal.

**Delete:** `window.confirm('Delete this note?')` -> `deleteNote(token, note.id)` -> `loadNotes()`.

**closeModal helper:** Sets `showCreateModal(false)`, `setEditNote(null)`, resets formData.

**Render structure (inline styles):**
